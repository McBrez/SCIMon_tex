% --------------------------------------------------------------------- Configuration Management --
% The value of target_document defines which document shall be build. Following values are 
% available:
% 0 ... research proposal
% 1 ... thesis
\newcounter{target_document}
\setcounter{target_document}{1}
\ifcase\value{target_document}
	\input{tex/research_proposal/research_proposal.tex}
\or
	\input{tex/thesis/thesis.tex}
\else
	\begin{document}
		Unknown target. Check the value of the latex variable target\_document.
	\end{document}
\fi
